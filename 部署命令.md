# 服务器部署命令

## 方法一：优先 git pull（推荐）

```bash
cd /opt/interview-master && git pull && cd frontend && npm run build && systemctl reload nginx
```

其中 `git pull` 可能会因 GitHub 连接失败而报错，如果失败用方法二。

---

## 方法二：手动用 sed 改文件（无需 SSH 传输，在服务器本地修改）

以下命令在服务器上直接修改 `QuestionGrid.tsx`，把 `onChange` 方案改为 `scroll` 事件方案：

### 第一步：删除旧代码

```bash
cd /opt/interview-master/frontend/src/components && cp QuestionGrid.tsx QuestionGrid.tsx.bak
```

### 第二步：用 cat 覆盖整个文件

```bash
cat > /opt/interview-master/frontend/src/components/QuestionGrid.tsx << 'ENDOFFILE'
// ============================================================
// QuestionGrid — 虚拟滚动卡片网格 + 无限滚动加载
// ============================================================

import { useCallback, useEffect, useRef } from 'react'
import { Empty, Spin } from 'antd'
import { useVirtualizer } from '@tanstack/react-virtual'
import Flashcard from './Flashcard'
import { useAppStore } from '../store'

const COLUMNS = 3
const CARD_HEIGHT = 220
const LOAD_MORE_THRESHOLD = 800

export default function QuestionGrid() {
  const { filteredQuestions, loading, hasMore, loadMore } = useAppStore()

  const rowCount = Math.ceil(filteredQuestions.length / COLUMNS)
  const totalRef = useRef(filteredQuestions.length)
  const loadingMoreRef = useRef(false)

  const hasMoreRef = useRef(hasMore)
  hasMoreRef.current = hasMore
  const loadingRef = useRef(loading)
  loadingRef.current = loading

  const getRowItems = useCallback(
    (rowIndex: number) => {
      const start = rowIndex * COLUMNS
      return filteredQuestions.slice(start, start + COLUMNS)
    },
    [filteredQuestions],
  )

  const virtualizer = useVirtualizer({
    count: rowCount || 1,
    getScrollElement: () => document.documentElement,
    estimateSize: () => CARD_HEIGHT,
    overscan: 3,
  })

  useEffect(() => {
    const handleScroll = () => {
      if (!hasMoreRef.current || loadingRef.current || loadingMoreRef.current) return
      const el = document.documentElement
      const distanceFromBottom = el.scrollHeight - el.scrollTop - el.clientHeight

      if (distanceFromBottom < LOAD_MORE_THRESHOLD) {
        loadingMoreRef.current = true
        loadMore().finally(() => {
          loadingMoreRef.current = false
        })
      }
    }
    window.addEventListener('scroll', handleScroll, { passive: true })
    return () => window.removeEventListener('scroll', handleScroll)
  }, [loadMore])

  useEffect(() => {
    if (filteredQuestions.length !== totalRef.current) {
      const wasFiltered = totalRef.current > 0 && filteredQuestions.length <= 50 && filteredQuestions.length < totalRef.current
      totalRef.current = filteredQuestions.length
      virtualizer.measure()
      if (wasFiltered) {
        window.scrollTo({ top: 0, behavior: 'smooth' })
      }
    }
  }, [filteredQuestions.length, virtualizer])

  if (loading && filteredQuestions.length === 0) {
    return (
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-5">
        {Array.from({ length: 6 }).map((_, i) => (
          <div key={i} className="h-[200px] rounded-xl bg-gray-100 animate-pulse" />
        ))}
      </div>
    )
  }

  if (filteredQuestions.length === 0) {
    return <Empty description="No matching questions" className="py-20" />
  }

  return (
    <div className="relative w-full">
      <div style={{ height: virtualizer.getTotalSize() }}>
        {virtualizer.getVirtualItems().map((virtualRow) => {
          const items = getRowItems(virtualRow.index)
          return (
            <div
              key={virtualRow.key}
              className="absolute top-0 left-0 w-full grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-5"
              style={{
                height: CARD_HEIGHT,
                transform: `translateY(${virtualRow.start}px)`,
              }}
            >
              {items.map((q) => (
                <Flashcard key={q.id} question={q} />
              ))}
            </div>
          )
        })}
      </div>
      {loading && filteredQuestions.length > 0 && (
        <div className="flex justify-center py-8">
          <Spin size="default" />
        </div>
      )}
      {!hasMore && filteredQuestions.length > 0 && (
        <p className="text-center text-gray-400 text-sm py-8">
          已展示全部 {filteredQuestions.length} 道题目
        </p>
      )}
    </div>
  )
}
ENDOFFILE
```

### 第三步：构建前端

```bash
cd /opt/interview-master/frontend && npm run build && systemctl reload nginx
```

### 第四步：验证

```bash
curl -s -o /dev/null -w '%{http_code}' http://127.0.0.1/
```
